// Initialize Reviews Section async function initReviewsSection(productId) { const reviewsSection = document.getElementById('reviewsSection'); const reviews = await getProductReviews(productId); if (reviewsSection) { reviewsSection.innerHTML = `

Avis des clients

Donner votre avis

${reviews.length > 0 ? reviews.map(review => createReviewCard(review)).join('') : `

Aucun avis pour le moment. Soyez le premier à donner votre avis!

`}
`; } } // Create review card HTML function createReviewCard(review) { const date = new Date(review.createdAt); const formattedDate = date.toLocaleDateString('fr-FR', { year: 'numeric', month: 'long', day: 'numeric' }); let starsHtml = ''; for (let i = 0; i < 5; i++) { if (i < review.rating) { starsHtml += ''; } else { starsHtml += ''; } } return `

${escapeHtml(review.userName)}

${formattedDate}
${starsHtml}

${escapeHtml(review.comment)}

`; } // Select star rating function selectStar(rating) { document.querySelectorAll('.star-input-container label').forEach((label, index) => { if (index < rating) { label.style.color = '#d4af37'; } else { label.style.color = '#ddd'; } }); } // Submit review async function submitReview(event, productId) { event.preventDefault(); const selectedRating = document.querySelector('input[name="rating"]:checked'); if (!selectedRating) { showNotification('Veuillez sélectionner une note!', 'warning'); return; } const reviewerName = document.getElementById('reviewerName').value.trim(); const reviewText = document.getElementById('reviewText').value.trim(); const submitBtn = document.getElementById('submitReviewBtn'); submitBtn.disabled = true; submitBtn.innerHTML = ' Envoi en cours...'; // Add review await addProductReview(productId, { userName: reviewerName, rating: parseInt(selectedRating.value), comment: reviewText }); // Refresh reviews list await initReviewsSection(productId); // Refresh rating display await initRatingDisplay(productId); // Reset form document.getElementById('reviewForm').reset(); document.querySelectorAll('.star-input-container label').forEach(label => { label.style.color = '#ddd'; }); showNotification('Votre avis a été envoyé avec succès!'); submitBtn.disabled = false; submitBtn.innerHTML = ' Envoyer mon avis'; } // Escape HTML to prevent XSS function escapeHtml(text) { const div = document.createElement('div'); div.textContent = text; return div.innerHTML; }
